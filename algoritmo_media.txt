Algoritmo <Média Aritmética>

Var

CARACTERE: nome;
REAL: notaUm, notaDois, notaTres, soma, media;

Escreva("Insira o nome do aluno");
Leia(nome);

Escreva("Insira a nota obtida na primeira prova");
Leia(notaUm);

Escreva("Insira a nota obtida na segunda prova");
Leia(notaDois);

Escreva("Insira a nota obtida na terceira prova");
Leia(notaTres);

soma <- notaUm + notaDois + notaTres;

media <- soma / 3

Escreva("O nome do Aluno e sua média obtida são:"+ nome "," + media);
